-- Table structure for `results`
CREATE TABLE `results` (
  `id` int NOT NULL AUTO_INCREMENT,
  `user_id` int NOT NULL,
  `semester` varchar(20) COLLATE utf8mb4_0900_ai_ci NOT NULL,
  `subject_1` int NOT NULL,
  `subject_2` int NOT NULL,
  `subject_3` int NOT NULL,     
  `subject_4` int NOT NULL,
  `subject_5` int NOT NULL,
  `extra_curricular` int NOT NULL,
  `total_marks` int NOT NULL,
  `created_at` datetime DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;


-- Table structure for `subjects`
CREATE TABLE `subjects` (
  `id` int NOT NULL AUTO_INCREMENT,
  `semester` varchar(50) COLLATE utf8mb4_0900_ai_ci NOT NULL,
  `subject_name` varchar(100) COLLATE utf8mb4_0900_ai_ci NOT NULL,
  `subject_code` varchar(20) COLLATE utf8mb4_0900_ai_ci DEFAULT NULL,
  `created_at` timestamp DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;


-- Table structure for `users`
CREATE TABLE `users` (
  `id` int NOT NULL AUTO_INCREMENT,
  `name` varchar(100) COLLATE utf8mb4_0900_ai_ci NOT NULL,
  `email` varchar(100) COLLATE utf8mb4_0900_ai_ci NOT NULL,
  `contact_no` varchar(15) COLLATE utf8mb4_0900_ai_ci NOT NULL,
  `alt_contact_no` varchar(15) COLLATE utf8mb4_0900_ai_ci NOT NULL,
  `current_year` varchar(20) COLLATE utf8mb4_0900_ai_ci NOT NULL,
  `current_sem` varchar(20) COLLATE utf8mb4_0900_ai_ci NOT NULL,
  `password` varchar(255) COLLATE utf8mb4_0900_ai_ci NOT NULL,
  `created_at` datetime DEFAULT CURRENT_TIMESTAMP,
  `updated_at` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;


-- Table structure for `admins`
CREATE TABLE `admins` (
  `id` int NOT NULL AUTO_INCREMENT,
  `name` varchar(100) COLLATE utf8mb4_0900_ai_ci NOT NULL,
  `email` varchar(100) COLLATE utf8mb4_0900_ai_ci NOT NULL,
  `password` varchar(255) COLLATE utf8mb4_0900_ai_ci NOT NULL,
  `created_at` datetime DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  UNIQUE KEY `email` (`email`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;


CREATE TABLE `monitoring_logs` (
  `id` int NOT NULL AUTO_INCREMENT,
  `user_id` int NOT NULL,
  `semester` varchar(20) NOT NULL,
  `study_hours` int NOT NULL DEFAULT 0,
  `attendance_percentage` decimal(5,2) NOT NULL DEFAULT 0.00,
  `extra_tutoring_sessions` int NOT NULL DEFAULT 0,
  `updated_at` datetime DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  FOREIGN KEY (`user_id`) REFERENCES `users`(`id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE `subjects` (
  `id` int NOT NULL AUTO_INCREMENT,
  `semester` varchar(50) NOT NULL, -- e.g., "Sem 1"
  `slot_number` int NOT NULL,      -- 1, 2, 3, 4, or 5 (to match results table)
  `subject_name` varchar(100) NOT NULL,
  `subject_code` varchar(20) DEFAULT NULL,
  `created_at` timestamp DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  -- This prevents accidentally adding two "Subject 1"s for the same semester
  UNIQUE KEY `unique_sem_slot` (`semester`, `slot_number`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
